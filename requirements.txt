<script>

const CATEGORIES = {
  chokepoint: { label: 'Chokepoint',         color: '#f85149' },
  vessel:     { label: 'Vessel incident',    color: '#ff9a3d' },
  port:       { label: 'Port',               color: '#fbcd5b' },
  conflict:   { label: 'Conflict / security', color: '#bc8cff' },
  policy:     { label: 'Policy / tariffs',   color: '#79c0ff' },
  labour:     { label: 'Labour',             color: '#56d364' },
  route:      { label: 'Route shift',        color: '#9d4edd' },
  industry:   { label: 'Industry signal',    color: '#8b949e' },
};
const SEVERITY = {
  critical: { color: '#f85149' }, high: { color: '#ff9a3d' },
  medium:   { color: '#fbcd5b' }, low:  { color: '#56d364' },
  info:     { color: '#79c0ff' },
};

let EVENTS = [];
let LAST_UPDATED = null;
let activeCategory = 'all';
let activeEventId = null;
const markerMap = {};

const map = L.map('map', {
  center: [25, 55], zoom: 3, minZoom: 2, maxZoom: 10,
  worldCopyJump: true, zoomControl: true,
}).setView([25, 55], 3);

L.tileLayer('https://cartodb-basemaps-{s}.global.ssl.fastly.net/dark_all/{z}/{x}/{y}@2x.png', {
  attribution: '&copy; OpenStreetMap &copy; CARTO',
  subdomains: 'abcd', maxZoom: 19,
}).addTo(map);

function makeMarker(ev) {
  const color = (SEVERITY[ev.severity] || SEVERITY.medium).color;
  const isCritical = ev.severity === 'critical';
  const icon = L.divIcon({
    className: 'marker-pulse',
    html: `<div class="marker-dot ${isCritical ? 'critical' : ''}" style="background:${color}; color:${color};"></div>`,
    iconSize: [18, 18], iconAnchor: [9, 9],
  });
  const m = L.marker([ev.lat, ev.lng], { icon, title: ev.title });
  m.bindPopup(`
    <div><strong>${escapeHtml(ev.title)}</strong></div>
    <div style="margin-top:4px; color:#8b949e; font-size:11px;">${escapeHtml(ev.location)} · ${escapeHtml(ev.dateLabel)}</div>
    <div style="margin-top:8px;">${escapeHtml(ev.description)}</div>
    <div style="margin-top:8px; font-size:11px;"><a href="${ev.sourceUrl}" target="_blank" rel="noopener">${escapeHtml(ev.sourceLabel)} ↗</a></div>
  `, { maxWidth: 320 });
  m.on('click', () => {
    activeEventId = ev.id;
    renderFeed();
    const el = document.querySelector(`.feed-item[data-id="${ev.id}"]`);
    if (el) el.scrollIntoView({ behavior: 'smooth', block: 'center' });
  });
  return m;
}

function escapeHtml(s) {
  return String(s || '').replace(/[&<>"']/g, c => ({
    '&':'&amp;', '<':'&lt;', '>':'&gt;', '"':'&quot;', "'":'&#39;'
  }[c]));
}

function renderMarkers() {
  Object.values(markerMap).forEach(m => map.removeLayer(m));
  Object.keys(markerMap).forEach(k => delete markerMap[k]);
  filteredEvents().forEach(ev => {
    const m = makeMarker(ev);
    m.addTo(map);
    markerMap[ev.id] = m;
  });
}

function renderFilters() {
  const counts = { all: EVENTS.length };
  EVENTS.forEach(e => { counts[e.category] = (counts[e.category] || 0) + 1; });
  const buildChip = (cat, label, color) => {
    const count = counts[cat] || 0;
    const dot = color ? `<span class="chip-dot" style="background:${color};"></span>` : '';
    return `<button class="chip ${activeCategory === cat ? 'active' : ''}" data-cat="${cat}">${dot}<span>${label}</span><span class="chip-count">${count}</span></button>`;
  };
  let html = buildChip('all', 'All events', null);
  Object.entries(CATEGORIES).forEach(([k, v]) => { if (counts[k]) html += buildChip(k, v.label, v.color); });
  const node = document.getElementById('filters');
  node.innerHTML = html;
  node.querySelectorAll('.chip').forEach(chip => {
    chip.addEventListener('click', () => {
      activeCategory = chip.dataset.cat;
      activeEventId = null;
      renderAll();
    });
  });
}

function filteredEvents() {
  const list = activeCategory === 'all' ? EVENTS : EVENTS.filter(e => e.category === activeCategory);
  const sevOrder = { critical: 5, high: 4, medium: 3, low: 2, info: 1 };
  return [...list].sort((a, b) => {
    if (a.date !== b.date) return b.date.localeCompare(a.date);
    return (sevOrder[b.severity] || 0) - (sevOrder[a.severity] || 0);
  });
}

function renderFeed() {
  const list = filteredEvents();
  document.getElementById('feed-count').textContent = list.length;
  document.getElementById('event-count').textContent = `${EVENTS.length} events`;
  const container = document.getElementById('feed-list');
  if (!list.length) {
    container.innerHTML = '<div class="empty-state">No events match this filter.</div>';
    return;
  }
  container.innerHTML = list.map(ev => {
    const sev = SEVERITY[ev.severity] || SEVERITY.medium;
    const cat = CATEGORIES[ev.category] || CATEGORIES.industry;
    return `
      <div class="feed-item ${activeEventId === ev.id ? 'active' : ''}" data-id="${ev.id}">
        <div class="feed-item-header">
          <span class="severity-dot" style="background:${sev.color};"></span>
          <span class="feed-item-cat" style="color:${cat.color}">${cat.label}</span>
          <span class="feed-item-date">${escapeHtml(ev.dateLabel)}</span>
        </div>
        <div class="feed-item-title">${escapeHtml(ev.title)}</div>
        <div class="feed-item-location">📍 ${escapeHtml(ev.location)}</div>
        <div class="feed-item-desc">${escapeHtml(ev.description)}</div>
        <div class="feed-item-source"><a href="${ev.sourceUrl}" target="_blank" rel="noopener">${escapeHtml(ev.sourceLabel)} ↗</a></div>
      </div>
    `;
  }).join('');
  container.querySelectorAll('.feed-item').forEach(item => {
    item.addEventListener('click', (e) => {
      if (e.target.tagName === 'A') return;
      const id = item.dataset.id;
      const ev = EVENTS.find(x => x.id === id);
      if (!ev) return;
      activeEventId = id;
      map.flyTo([ev.lat, ev.lng], 5, { duration: 0.8 });
      setTimeout(() => markerMap[id]?.openPopup(), 850);
      renderFeed();
    });
  });
}

function renderAll() {
  renderFilters();
  renderMarkers();
  renderFeed();
}

function formatUpdated(iso) {
  if (!iso) return 'unknown';
  const d = new Date(iso);
  const now = new Date();
  const diffMin = Math.round((now - d) / 60000);
  if (diffMin < 1) return 'just now';
  if (diffMin < 60) return `${diffMin}m ago`;
  if (diffMin < 1440) return `${Math.round(diffMin/60)}h ago`;
  return d.toLocaleDateString(undefined, { month: 'short', day: 'numeric' });
}

async function loadEvents() {
  try {
    const url = `events.json?t=${Date.now()}`; // cache-buster
    const response = await fetch(url);
    if (!response.ok) throw new Error(`HTTP ${response.status}`);
    const data = await response.json();
    EVENTS = data.events || [];
    LAST_UPDATED = data.lastUpdated;
    document.getElementById('status').innerHTML =
      `<span class="pulse"></span>&nbsp;&nbsp;Updated ${formatUpdated(LAST_UPDATED)}`;
    renderAll();
  } catch (e) {
    console.error('Failed to load events:', e);
    document.getElementById('status').textContent = 'Failed to load events.json';
    document.getElementById('feed-list').innerHTML =
      `<div class="empty-state">Could not load <code>events.json</code>.<br/><br/>` +
      `If you opened this file directly (file://), browsers block fetch for security.<br/>` +
      `Try: <code>python -m http.server</code> from the project folder, then visit ` +
      `<code>http://localhost:8000</code>.<br/><br/>` +
      `Or deploy to GitHub Pages.</div>`;
  } finally {
    document.getElementById('loading').classList.add('hidden');
  }
}

loadEvents();
</script>
